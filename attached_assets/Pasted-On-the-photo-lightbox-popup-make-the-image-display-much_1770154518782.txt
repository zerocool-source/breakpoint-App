**4. ZOOM FUNCTIONALITY:**
```javascript
let zoomLevel = 100;
let isDragging = false;
let startX, startY, scrollLeft, scrollTop;

// Zoom in - increase by 50% each click for high-res images
function zoomIn() {
  zoomLevel = Math.min(zoomLevel + 50, 400); // Allow up to 400%
  applyZoom();
}

// Zoom out
function zoomOut() {
  zoomLevel = Math.max(zoomLevel - 50, 50);
  applyZoom();
}

// Reset to fit screen
function resetZoom() {
  zoomLevel = 100;
  applyZoom();
}

// Apply zoom transform
function applyZoom() {
  const image = document.querySelector('.lightbox-image');
  image.style.transform = `scale(${zoomLevel / 100})`;
  
  // Update zoom display
  document.querySelector('.zoom-level').textContent = `${zoomLevel}%`;
  
  // Enable dragging when zoomed past 100%
  if (zoomLevel > 100) {
    image.style.cursor = 'grab';
  } else {
    image.style.cursor = 'zoom-in';
  }
}

// Mouse wheel zoom
function handleWheel(e) {
  e.preventDefault();
  if (e.deltaY < 0) {
    zoomIn();
  } else {
    zoomOut();
  }
}

// Double-click to toggle zoom
function handleDoubleClick() {
  if (zoomLevel === 100) {
    zoomLevel = 200;
  } else {
    zoomLevel = 100;
  }
  applyZoom();
}
```

**5. IMAGE SHOULD FILL SCREEN IMMEDIATELY:**
