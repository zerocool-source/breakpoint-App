When a work order is converted, update its status in the database:

```javascript
// When converting to Estimate
const convertToEstimate = async (workOrder) => {
  // 1. Create the estimate
  const estimate = await createEstimate(workOrder);
  
  // 2. Mark the work order as converted (so it doesn't show again)
  await updateWorkOrder(workOrder.id, {
    status: 'converted_to_estimate',
    converted_at: new Date(),
    estimate_id: estimate.id
  });
  
  // 3. Remove from UI list
  setWorkOrders(prev => prev.filter(wo => wo.id !== workOrder.id));
  
  // 4. Update count badge
  setWorkOrderCount(prev => prev - 1);
};

// When converting to Invoice
const convertToInvoice = async (selectedWorkOrders) => {
  for (const workOrder of selectedWorkOrders) {
    // 1. Create the invoice
    const invoice = await createInvoice(workOrder);
    
    // 2. Mark the work order as invoiced
    await updateWorkOrder(workOrder.id, {
      status: 'invoiced',
      converted_at: new Date(),
      invoice_id: invoice.id
    });
  }
  
  // 3. Remove from UI list
  setWorkOrders(prev => 
    prev.filter(wo => !selectedWorkOrders.find(s => s.id === wo.id))
  );
  
  // 4. Update count badge
  setWorkOrderCount(prev => prev - selectedWorkOrders.length);
  
  // 5. Clear selection
  setSelectedWorkOrders([]);
};
```

**4. FILTER OUT CONVERTED WORK ORDERS WHEN FETCHING:**
```javascript
// When fetching work orders for the container
const fetchWorkOrders = async () => {
  const response = await fetch('/api/work-orders?status=pending');
  // Only show work orders that haven't been converted
  // Filter where status is NOT 'converted_to_estimate' AND NOT 'invoiced'
  const workOrders = await response.json();
  return workOrders.filter(wo => 
    wo.status !== 'converted_to_estimate' && 
    wo.status !== 'invoiced'
  );
};
```
